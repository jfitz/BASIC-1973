90 REM * ALPHABETIZE NAMES IN A FILE
100 DIM A$[72], B$[72]
110 FILES T E S T
120 FOR I = 1 TO 6
130 READ #1, I; A$, A, A1
140 FOR J = I + 1 TO 7
150 READ #1, J; B$, B, B1
160 IF A$ <= B$ THEN 200
170 PRINT #1, I; B$, B, B1
180 PRINT #1, J; A$, A, A1
190 READ #1, I; A$, A, A1
200 NEXT J
210 NEXT I
220 PRINT "FILE ALPHABETIZED"
230 END
